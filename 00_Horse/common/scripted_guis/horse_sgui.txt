
is_trot = {
	scope = character

	is_shown = {
		has_character_flag = is_trotting
	}

	effect = {
		if = {
			limit = {
				has_character_flag = is_trotting
			}
			remove_character_flag = is_trotting
		}
		else = {
			add_character_flag = is_trotting
		}
		
	} 
}

is_not_trot = {
	scope = character
	is_shown = {
		NOT = {
			has_character_flag = is_trotting
		}
	}
	effect = {
		remove_character_flag = is_trotting
	} 
}

trot_l = {

    scope = character

	effect = {
		if = {
			limit = {
				has_character_flag = is_trotting
			}
			remove_character_flag = horse_ridingr
			add_character_flag = horse_ridingl
		}
		else = {
			remove_character_flag = horse_ridingl
			remove_character_flag = horse_ridingr
		}

	}
}
