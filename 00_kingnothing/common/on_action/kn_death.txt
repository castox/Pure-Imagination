# character just about to die in root scope
# if a killer is know, it's set as scope:killer
#Triggered by code
on_death = {
	effect = {
		if = { 
			limit = { 
				AND = {
					has_character_modifier = king_nothing_modifier
					exists = player_heir
				}
			}
			player_heir = {
				add_character_modifier = { modifier = king_nothing_modifier } 
			}
		}
	}
}
