new_castox = {
	create_dynamic_title = {
		tier = empire
		name = "Merchanchant Empire"
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	scope:new_title = {
		set_capital_county = title:c_naxos
		set_landless_title = yes
		set_destroy_on_succession = yes
		set_delete_on_destroy = yes
		set_no_automatic_claims = yes
		set_definitive_form = yes
		set_can_be_named_after_dynasty = no
		change_title_holder = {
			holder = global_var:castox
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
	global_var:castox = {
		set_variable = {
			name = peasant_title
			value = scope:new_title
		}
	}
	scope:new_title = { generate_coa = factions }
}
reincarnate_castox = {
	if = {
		limit = {
			global_var:castox = { is_alive = no }
		}

			create_character = {
				template_character = global_var:castox
				save_scope_as = new_castox
				template = castox_template
				location = title:c_naxos
				after_creation = {
					copy_inheritable_appearance_from = global_var:castox
					set_immortal_age = 24
					change_first_name = "Castox"
					set_house = global_var:castox.house
					add_character_flag = has_scripted_appearance
					set_global_variable = { name = castox value = scope:new_castox }
					new_castox = yes
				}
			}
		
	}
}
