
### COMBAT - DUELING PERKS


# Windstance
windstance_perk = {
	lifestyle = martial_lifestyle
	tree = dueling
	position = { 1 0 }
	icon = node_combat
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_martial_trigger = yes
			}
			add = 989
			if = {
				limit = {
					OR = {
						has_trait = ambitious
						has_trait = arrogant
						has_trait = brave
					}
					ai_boldness > 0
					ai_energy > 0
					prowess > 10
					age < 30
				}
				add = 500
			}
		}
		if = {
			limit = {
				has_any_shard_artifacts = yes
			}
			every_in_list = {
				variable = artifact_list
				add = 100
			}
			every_in_list = {
				variable = artifact_list_shard
				add = 1000
			}
		}
		if = {
			limit = {
				has_trait = duelist
			}
			multiply = 10
		}
		if = {
			limit = {
				has_trait = blademaster_1
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = blademaster_2
			}
			multiply = 3
		}
		if = {
			limit = {
				has_trait = blademaster_3
			}
			multiply = 4
		}
		if = {
			limit = {
				OR = {
					has_council_position = kingsguard_1
					has_council_position = kingsguard_2
					has_council_position = kingsguard_3
					has_council_position = kingsguard_4
					has_council_position = kingsguard_5
					has_council_position = kingsguard_6
				}
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}
	
	character_modifier = {
		prowess = 3
		movement_speed = 0.1
	}
	
	effect = {
		custom_description_no_bullet = {
			text = windstance_perk_effect
		}
	}
}
